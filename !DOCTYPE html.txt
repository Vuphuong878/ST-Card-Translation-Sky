    function formatNum(num) {

        const parsed = parseInt(num, 10);
        return isNaN(parsed) ? '0' : parsed.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
    }

    function bindEvents() {
      $('.section-header').off('click').on('click', function() {
        $(this).toggleClass('collapsed').next('.section-content').toggleClass('hidden');
      });

      $('.hm-item').off('click').on('click', function() {
        const name = $(this).data('name');
        const data = JSON.parse(decodeURIComponent($(this).data('details')));

        $('#m-name').text(name);
        $('#m-title').text(data['Vị Phân'] || 'Không Có');
        $('#m-age').text(`Tuổi Tác: ${data['Tuổi Tác']} Tuổi`);
        $('#m-fav').text(data['Hảo Cảm'] || 0);
        $('#m-sex').text(data['Số Lần Ân Ái'] || 0);
        $('#m-eval').text(data['Đánh Giá'] || 'Không Có');
        $('#m-desc').text(data['Giải Thích'] || 'Không có giải thích');

        /* Logic xử lý ảnh đứng (Luân phiên nhiều ảnh) */

        const pData = portraitsData[name];

        if (pData) {
            let normalList = pData.normal || [];
            let nsfwList = pData.nsfw || [];

            if (fav >= 61) {
                currentImageGallery = normalList.concat(nsfwList);
            } else {
                currentImageGallery = normalList;
            }

            currentImageIndex = 0;

            if (currentImageGallery.length > 0) {
                $('#m-portrait').attr('src', currentImageGallery[0]).show();
                $('#m-portrait-tip').show();

                canToggleImage = currentImageGallery.length > 1;

                if (canToggleImage) {
                    if (fav >= 61 && nsfwList.length > 0) {
                        $('#m-portrait-tip').text('✦ Nhấp vào bức họa để thay y phục ✦');
                    } else {
                        $('#m-portrait-tip').text('✦ Nhấp vào bức họa để xem thêm ✦ (Hảo cảm 61 có thể thay y phục)');
                    }
                    $('#m-portrait').css('cursor', 'pointer');
                } else {
                    $('#m-portrait-tip').text('✦ Bức họa giai nhân ✦ (Hảo cảm 61 có thể thay y phục)');
                    $('#m-portrait').css('cursor', 'default');
                }
            } else {
                $('#m-portrait').hide();
                $('#m-portrait-tip').hide();
            }
        } else {
            $('#m-portrait').hide();
            $('#m-portrait-tip').hide();
        }

        $('#harem-modal').fadeIn(200).css('display', 'flex');
      });

      $('#m-portrait').off('click').on('click', function() {
          if (canToggleImage && currentImageGallery.length > 1) {
              currentImageIndex = (currentImageIndex + 1) % currentImageGallery.length;
              /* Chuyển đổi hoạt ảnh độ mờ để tránh nhấp nháy cứng nhắc */
              $(this).css('opacity', 0.5);
              setTimeout(() => {
                  $(this).attr('src', currentImageGallery[currentImageIndex]).css('opacity', 1);
              }, 150);
          }
      });

      $('.modal-close, .modal-overlay').off('click').on('click', function(e) {
        if (e.target === this) $('#harem-modal').fadeOut(200);
      });
    }
